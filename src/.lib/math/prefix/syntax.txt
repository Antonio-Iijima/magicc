<MATHEXPR>   ::= <ADD> | <SUBTRACT> | <TERM> | <COMPARISON>
<ADD>        ::= \+ <MATHEXPR> <TERM>
<SUBTRACT>   ::= - <MATHEXPR> <TERM>
<TERM>       ::= <MULTIPLY> | <DIVIDE> | <FACTOR>
<MULTIPLY>   ::= \* <TERM> <FACTOR>
<DIVIDE>     ::= / <TERM> <FACTOR>
<FACTOR>     ::= \( <MATHEXPR> \) | <VALUE>
<VALUE>      ::= <NUMBER>

<COMPARISON> ::= <LE> | <LEQ> | <GE> | <GEQ> | <EQ> | <NEQ>
<LE>         ::= < <MATHEXPR> <MATHEXPR>
<LEQ>        ::= <= <MATHEXPR> <MATHEXPR>
<GE>         ::= > <MATHEXPR> <MATHEXPR>
<GEQ>        ::= >= <MATHEXPR> <MATHEXPR>
<EQ>         ::= == <MATHEXPR> <MATHEXPR>
<NEQ>        ::= != <MATHEXPR> <MATHEXPR>
